Step 2: Copy This Script

On that Supabase page, press F12 → click Console tab → paste this code and press Enter:

(function() { // Click all "Reveal" buttons first document.querySelectorAll('button').forEach(b => { if (b.textContent.includes('Reveal') || b.textContent.includes('reveal')) b.click(); }); setTimeout(() => { const keys = {}; // Try inputs document.querySelectorAll('input').forEach(inp => { const v = inp.value || ''; if (v.startsWith('eyJ') && v.length > 100) { const label = inp.getAttribute('aria-label') || inp.closest('[class*="row"]')?.querySelector('p,span,label')?.textContent || 'key'; keys[label] = v; } }); // Try textareas document.querySelectorAll('textarea').forEach(ta => { const v = ta.value || ''; if (v.startsWith('eyJ') && v.length > 100) { keys['textarea'] = v; } }); // Try visible text const allText = document.body.innerText; const matches = allText.match(/eyJ[A-Za-z0-9_-]+\.[A-Za-z0-9_-]+\.[A-Za-z0-9_-]+/g) || []; matches.forEach((m, i) => { if (m.length > 100) keys['found_' + i] = m; }); const result = JSON.stringify(keys, null, 2); console.log('%cSUPABASE KEYS:', 'color:lime;font-size:20px;font-weight:bold'); console.log(result); // Try to send to our local listener fetch('http://localhost:7777/keys', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: result }).then(() => console.log('%cKeys sent to local listener!', 'color:cyan')).catch(() => console.log('Local listener not running')); return result; }, 1000); })();
